#include<unordered_map>
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> hashTab;
       for(int i = 0;i< nums.size(); ++i)
       {
           auto it = hashTab.find(target - nums[i]);
           if(it != hashTab.end())
           {
               return {it->second, i};
           }
            hashTab.emplace(nums[i], i);
       }
       return {};
    }
};
//runtime:8 ms
//memory:10.7 MB
